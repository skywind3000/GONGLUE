按 TAB 键後再输入密技
allammo = 弹药全满
behindview1 = 第三人称视角
behindview0 =第一人称视角
flush = 重绘图素
fly = 飞行模式
ghost = 穿墙模式
god = 无敌模式
hideactors = 隐藏所有怪物,武器和物品
showactors = 显示所有怪物,武器和物品
invisible0 = 关闭隐形模式
invisible1 = 开启隐形模式
killall # = 杀死特定怪物 (#=怪物名称)
killpawns = 杀死所有怪物
open # = 跳关 (#=关卡名称)
playersonly = 计时器暂停
slomo # =设定游戏时序 (#=1 为正常)
summon # = 得到武器,物品或怪物 (#=武器,物品或怪物名称)
suicide = 自杀
walk =由飞行或穿墙模式中回复
